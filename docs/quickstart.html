Quickstart

Get up and running with Agents Hub in minutes.

Installation

Install Agents Hub using pip:

pip install agents-hub

For development or to use the latest features, you can install directly from GitHub:

pip install git+https://github.com/emaginest/agents-hub.git

Prerequisites

To use Agents Hub, you'll need:

  • Python 3.9 or higher
  • API keys for the LLM providers you want to use (OpenAI, Anthropic, Google, etc.)

Note: You can use Ollama for local LLMs without an API key.

Creating a Basic Agent

Let's start by creating a simple agent using the OpenAI provider:

import os
import asyncio
from agents_hub import Agent
from agents_hub.llm.providers import OpenAIProvider

# Initialize LLM provider
llm = OpenAIProvider(
    api_key=os.environ.get("OPENAI_API_KEY"),
    model="gpt-4o"  # You can use other models like "gpt-3.5-turbo"
)

# Create an agent
agent = Agent(
    name="assistant",
    llm=llm,
    system_prompt="You are a helpful assistant."
)

# Use the agent
async def main():
    response = await agent.run("What is the capital of France?")
    print(response)

# Run the async function
asyncio.run(main())

Using Different LLM Providers

Agents Hub supports multiple LLM providers. Here's how to use Claude:

from agents_hub.llm.providers import ClaudeProvider

# Initialize Claude provider
llm = ClaudeProvider(
    api_key=os.environ.get("ANTHROPIC_API_KEY"),
    model="claude-3-opus-20240229"  # You can use other models like "claude-3-sonnet-20240229"
)

# Create an agent with Claude
agent = Agent(
    name="assistant",
    llm=llm,
    system_prompt="You are a helpful assistant."
)

Or use Ollama for local LLMs:

from agents_hub.llm.providers import OllamaProvider

# Initialize Ollama provider
llm = OllamaProvider(
    model="llama3",  # Or any other model you have in Ollama
    base_url="http://localhost:11434"  # Default Ollama URL
)

# Create an agent with Ollama
agent = Agent(
    name="assistant",
    llm=llm,
    system_prompt="You are a helpful assistant."
)

Adding Tools to an Agent

Let's enhance our agent with tools for web search and calculations:

from agents_hub import Agent
from agents_hub.llm.providers import OpenAIProvider
from agents_hub.tools.standard import WebSearchTool, CalculatorTool

# Initialize LLM provider
llm = OpenAIProvider(api_key=os.environ.get("OPENAI_API_KEY"))

# Initialize tools
web_search = WebSearchTool()
calculator = CalculatorTool()

# Create an agent with tools
agent = Agent(
    name="research_assistant",
    llm=llm,
    tools=[web_search, calculator],
    system_prompt="You are a research assistant that can search the web and perform calculations."
)

async def main():
    response = await agent.run("What is the population of Tokyo and calculate it as a percentage of Japan's population?")
    print(response)

asyncio.run(main())

Creating an Agent Workforce

Now let's create a team of specialized agents that work together:

from agents_hub import AgentWorkforce, Agent
from agents_hub.llm.providers import OpenAIProvider, ClaudeProvider

# Initialize LLM providers
openai_llm = OpenAIProvider(api_key=os.environ.get("OPENAI_API_KEY"))
claude_llm = ClaudeProvider(api_key=os.environ.get("ANTHROPIC_API_KEY"))

# Create specialized agents
researcher = Agent(
    name="researcher",
    llm=claude_llm,
    system_prompt="You are a researcher who finds and analyzes information."
)

analyst = Agent(
    name="analyst",
    llm=openai_llm,
    system_prompt="You are an analyst who evaluates information and draws insights."
)

writer = Agent(
    name="writer",
    llm=claude_llm,
    system_prompt="You are a writer who creates clear and engaging content."
)

# Create a workforce with these agents
workforce = AgentWorkforce(
    agents=[researcher, analyst, writer],
    orchestrator_agent=analyst  # The analyst will coordinate the workflow
)

async def main():
    # Execute a task with the workforce
    result = await workforce.execute(
        "Research the impact of AI on healthcare and prepare a report",
        context={
            "format": "markdown",
            "max_length": 1000,
            "focus_areas": ["diagnosis", "treatment", "patient care"]
        }
    )

    print(result["result"])

asyncio.run(main())

Creating a Cognitive Agent

Let's create an agent with advanced cognitive capabilities:

from agents_hub import CognitiveAgent
from agents_hub.llm.providers import ClaudeProvider
from agents_hub.cognitive import CognitiveArchitecture

# Initialize LLM provider
llm = ClaudeProvider(api_key=os.environ.get("ANTHROPIC_API_KEY"))

# Create cognitive architecture
cognitive_architecture = CognitiveArchitecture(
    metacognition_config={
        "reflection_depth": 2,
        "confidence_threshold": 0.7,
    },
    reasoning_config={
        "enabled_mechanisms": ["deductive", "inductive", "abductive"],
        "default_mechanism": "deductive",
    },
)

# Create cognitive agent
agent = CognitiveAgent(
    name="cognitive_agent",
    llm=llm,
    cognitive_architecture=cognitive_architecture,
    system_prompt="You are a thoughtful assistant with advanced reasoning capabilities.",
    cognitive_config={
        "reasoning_trace_enabled": True,
        "metacognition_enabled": True,
        "learning_enabled": True,
    },
)

async def main():
    # Use the cognitive agent with different reasoning mechanisms
    response = await agent.run(
        "What are the main causes of climate change?",
        context={"reasoning_mechanism": "deductive"}
    )

    print(response)

    # You can also access the reasoning trace
    print("\nReasoning Trace:")
    print(agent.last_reasoning_trace)

asyncio.run(main())

Adding Memory to an Agent

Let's create an agent with a memory system:

from agents_hub import Agent
from agents_hub.llm.providers import OpenAIProvider
from agents_hub.memory.backends import PostgreSQLMemory

# Initialize LLM provider
llm = OpenAIProvider(api_key=os.environ.get("OPENAI_API_KEY"))

# Initialize memory backend
memory = PostgreSQLMemory(
    llm=llm,  # For generating embeddings
    host="localhost",
    port=5432,
    database="agents_hub",
    user="postgres",
    password="postgres"
)

# Create an agent with memory
agent = Agent(
    name="memory_agent",
    llm=llm,
    memory=memory,
    system_prompt="You are an assistant that remembers past conversations."
)

async def main():
    # First interaction
    response1 = await agent.run("My name is Alice and I'm interested in quantum computing.")
    print("Response 1:", response1)

    # Second interaction (agent should remember the user's name and interest)
    response2 = await agent.run("What topics was I interested in again?")
    print("Response 2:", response2)

asyncio.run(main())
